router.get("/", (req, res) => {
  const reviewsSelectQuery = `
          SELECT    id,
                    score,
                    title,
                    content,
                    createdAt,
                    UserId
            FROM    reviews
      `;
  try {
    db.query(reviewsSelectQuery, (error, reviews) => {
      console.log(reviews);

      res.render("screens/review", { reviews });
    });
  } catch (error) {
    console.log(error);
    return res.redirect("/");
  }
});
